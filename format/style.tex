\def\coverpage{
  % no page number for the cover page
  \thispagestyle{empty}

  \vspace*{2cm}
  \begin{center}
    \LARGE {\bf \thesistitle}

    \vskip 1cm
    \large \authorname

    \vskip 1cm
    \large
  % both Fulfilment and Fulfillment are correct
    A Thesis Submitted in Partial Fulfilment \\
    of the Requirements for the Degree of \\
    \degree \\
    in \\
    \programme \\

    \vfill
    \large \institution \\
    \submitdate
  \end{center}

  \newpage

  \thispagestyle{empty}
  \vspace*{10mm}
  \begin{center}
    \vfill
    \large
    \underline{\bf Thesis Assessment Committee}
    \vskip 1.5cm
    \committee
    \vfill
  \end{center}
  \normalsize
  \newpage
}
